// Load cart and ensure every item has a quantity (default to 1) let cart = JSON.parse(localStorage.getItem('atoll_cart')) || []; function addToCart(name, price, stripeLink) { const existingItem = cart.find(item => item.name === name); if (existingItem) { existingItem.quantity += 1; } else { cart.push({ name, price, stripeLink: stripeLink || DEFAULT_STRIPE_LINK, quantity: 1 }); } localStorage.setItem('atoll_cart', JSON.stringify(cart)); updateCartUI(); showToast(`Added ${name} to cart! 🛍️`); } function updateQuantity(index, change) { cart[index].quantity += change; if (cart[index].quantity <= 0) { cart.splice(index, 1); } localStorage.setItem('atoll_cart', JSON.stringify(cart)); updateCartUI(); } function removeFromCart(index) { cart.splice(index, 1); localStorage.setItem('atoll_cart', JSON.stringify(cart)); updateCartUI(); } function updateCartUI() { const container = document.getElementById('cart-items-container'); const cartCount = document.getElementById('cart-count'); const cartTotal = document.getElementById('cart-total-price'); const checkoutBtn = document.getElementById('cart-checkout-btn'); container.innerHTML = ''; let total = 0; let totalItems = 0; if (cart.length === 0) { container.innerHTML = '

Your cart is empty.

'; checkoutBtn.setAttribute('href', '#'); checkoutBtn.setAttribute('onclick', "alert('Your cart is empty! Add items to checkout.')"); checkoutBtn.textContent = "Checkout via Stripe"; } else { cart.forEach((item, index) => { const itemTotal = item.price * item.quantity; total += itemTotal; totalItems += item.quantity; const itemHtml = `

${item.name}

£${item.price.toFixed(2)} ${item.quantity > 1 ? `x ${item.quantity}` : ''}

${item.quantity}
`; container.insertAdjacentHTML('beforeend', itemHtml); }); // Still fallback to the last added item's direct link for now const checkoutUrl = cart[cart.length - 1].stripeLink; checkoutBtn.setAttribute('href', checkoutUrl); checkoutBtn.removeAttribute('onclick'); checkoutBtn.textContent = `Checkout via Stripe (£${total.toFixed(2)})`; } cartCount.textContent = totalItems; cartTotal.textContent = `£${total.toFixed(2)}`; }
